/* qmlpvint.cpp */
QmlPvInt::QmlPvInt(QObject *parent)
    : QmlPvObject{parent}
{}

void QmlPvInt::setPvName(const QString &pvname)
{
    if (!pvname.isEmpty() && pvname != m_pv_name) {
        m_pv_name = pvname;
        Q_EMIT pvNameChanged(pvname);

        if (!m_caobject.isNull()) {
            m_caobject->closeChannel();
            m_caobject->deleteLater();
        }

        m_caobject = new QEInteger(pvname, this, Q_NULLPTR, 0);
        connect(m_caobject, &QCaObject::connectionChanged, this, &QmlPvInt::onConnectionChanged);
        connect(m_caobject,
                QOverload<const QVariant&, QCaAlarmInfo&, QCaDateTime&, const unsigned int&>::of(&QCaObject::dataChanged),
                this, &QmlPvInt::onDataChanged);
        m_caobject->subscribe();
    }
}

void QmlPvInt::setValue(const QVariant &value)
{
    if (value != m_value) {
        bool ok = false;
        int val = value.toInt(&ok);
        if (ok && !m_caobject.isNull() && m_caobject->getChannelIsConnected()) {
            m_caobject->writeIntegerValue(val);
        }
    }
}
